animal_data.head()

#group months by seasons that I'm separating as: 12-02 winter, 03-05 spring, 06-08 summer, 09-11 fall
#used apply method from: https://pandas.pydata.org/docs/reference/api/pandas.DataFrame.apply.html instead of a for loop
def separating_seasons(month):
    if month in ['12', '01', '02']:
        return 'Winter'
    elif month in ['03', '04', '05']:
        return 'Spring'
    elif month in ['06', '07', '08']:
        return 'Summer'
    elif month in ['09', '10', '11']:
        return 'Fall'
    
animal_data['Intake Season'] = animal_data['Intake Month'].apply(separating_seasons)

#code to set season order
season_order = ['Spring', 'Summer', 'Fall', 'Winter']

#code to group data by seasons and easily count the intakes
#used this documentation to change the order of the season info: https://pandas.pydata.org/docs/reference/api/pandas.DataFrame.reindex.html
intake_seasonal_counts = animal_data['Intake Season'].value_counts().reindex(season_order)
